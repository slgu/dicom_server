174 void HttpRequest::decodeRequestParams() {
175  #ifdef SUPERVERBOSE
176  qDebug("HttpRequest: extract and decode request parameters");
177  #endif
178  // Get URL parameters
179  QByteArray rawParameters;
180  int questionMark=path.indexOf('?');
181  if (questionMark>=0) {
182  rawParameters=path.mid(questionMark+1);
183  path=path.left(questionMark);
184  }
185  // Get request body parameters
186  QByteArray contentType=headers.value("Content-Type");
187  if (!bodyData.isEmpty() && (contentType.isEmpty() || contentType.startsWith("application/x-www-form-urlencoded"))) {
188  if (!rawParameters.isEmpty()) {
189  rawParameters.append('&');
190  rawParameters.append(bodyData);
191  }
192  else {
193  rawParameters=bodyData;
194  }
195  }
196  // Split the parameters into pairs of value and name
197  QList<QByteArray> list=rawParameters.split('&');
198  foreach (QByteArray part, list) {
199  int equalsChar=part.indexOf('=');
200  if (equalsChar>=0) {
201  QByteArray name=part.left(equalsChar).trimmed();
202  QByteArray value=part.mid(equalsChar+1).trimmed();
203  parameters.insert(urlDecode(name),urlDecode(value));
204  }
205  else if (!part.isEmpty()){
206  // Name without value
207  parameters.insert(urlDecode(part),"");
208  }
209  }

311 QByteArray HttpRequest::urlDecode(const QByteArray source) {
312  QByteArray buffer(source);
313  buffer.replace('+',' ');
314  int percentChar=buffer.indexOf('%');
315  while (percentChar>=0) {
316  bool ok;
317  char byte=buffer.mid(percentChar+1,2).toInt(&ok,16);
318  if (ok) {
319  buffer.replace(percentChar,3,(char*)&byte,1);
320  }
321  percentChar=buffer.indexOf('%',percentChar+1);
322  }

327 void HttpRequest::parseMultiPartFile() {
328  qDebug("HttpRequest: parsing multipart temp file");
329  tempFile.seek(0);
330  bool finished=false;
331  while (!tempFile.atEnd() && !finished && !tempFile.error()) {
332  #ifdef SUPERVERBOSE
333  qDebug("HttpRequest: reading multpart headers");
334  #endif
335  QByteArray fieldName;
336  QByteArray fileName;
337  while (!tempFile.atEnd() && !finished && !tempFile.error()) {
338  QByteArray line=tempFile.readLine(65536).trimmed();
339  if (line.startsWith("Content-Disposition:")) {
340  if (line.contains("form-data")) {
341  int start=line.indexOf(" name=\"");
342  int end=line.indexOf("\"",start+7);
343  if (start>=0 && end>=start) {
344  fieldName=line.mid(start+7,end-start-7);
345  }
346  start=line.indexOf(" filename=\"");
347  end=line.indexOf("\"",start+11);
348  if (start>=0 && end>=start) {
349  fileName=line.mid(start+11,end-start-11);
350  }
351  #ifdef SUPERVERBOSE
352  qDebug("HttpRequest: multipart field=%s, filename=%s",fieldName.data(),fileName.data());
353  #endif
354  }
355  else {
356  qDebug("HttpRequest: ignoring unsupported content part %s",line.data());
357  }
358  }
359  else if (line.isEmpty()) {
360  break;
361  }
362  }
363 
364  #ifdef SUPERVERBOSE
365  qDebug("HttpRequest: reading multpart data");
366  #endif
367  QTemporaryFile* uploadedFile=0;
368  QByteArray fieldValue;
369  while (!tempFile.atEnd() && !finished && !tempFile.error()) {
370  QByteArray line=tempFile.readLine(65536);
371  if (line.startsWith("--"+boundary)) {
372  // Boundary found. Until now we have collected 2 bytes too much,
373  // so remove them from the last result
374  if (fileName.isEmpty() && !fieldName.isEmpty()) {
375  // last field was a form field
376  fieldValue.remove(fieldValue.size()-2,2);
377  parameters.insert(fieldName,fieldValue);
378  qDebug("HttpRequest: set parameter %s=%s",fieldName.data(),fieldValue.data());
379  }
380  else if (!fileName.isEmpty() && !fieldName.isEmpty()) {
381  // last field was a file
382  #ifdef SUPERVERBOSE
383  qDebug("HttpRequest: finishing writing to uploaded file");
384  #endif
385  uploadedFile->resize(uploadedFile->size()-2);
386  uploadedFile->flush();
387  uploadedFile->seek(0);
388  parameters.insert(fieldName,fileName);
389  qDebug("HttpRequest: set parameter %s=%s",fieldName.data(),fileName.data());
390  uploadedFiles.insert(fieldName,uploadedFile);
391  qDebug("HttpRequest: uploaded file size is %i",(int) uploadedFile->size());
392  }
393  if (line.contains(boundary+"--")) {
394  finished=true;
395  }
396  break;
397  }
398  else {
399  if (fileName.isEmpty() && !fieldName.isEmpty()) {
400  // this is a form field.
401  currentSize+=line.size();
402  fieldValue.append(line);
403  }
404  else if (!fileName.isEmpty() && !fieldName.isEmpty()) {
405  // this is a file
406  if (!uploadedFile) {
407  uploadedFile=new QTemporaryFile();
408  uploadedFile->open();
409  }
410  uploadedFile->write(line);
411  if (uploadedFile->error()) {
412  qCritical("HttpRequest: error writing temp file, %s",qPrintable(uploadedFile->errorString()));
413  }
414  }
415  }
416  }
417  }
418  if (tempFile.error()) {
419  qCritical("HttpRequest: cannot read temp file, %s",qPrintable(tempFile.errorString()));
420  }
421  #ifdef SUPERVERBOSE
422  qDebug("HttpRequest: finished parsing multipart temp file");
423  #endif
424 }
