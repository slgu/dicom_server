164 void HttpConnectionHandler::read() {
165  // The loop adds support for HTTP pipelinig
166  while (socket.bytesAvailable()) {
167  #ifdef SUPERVERBOSE
168  qDebug("HttpConnectionHandler (%p): read input",this);
169  #endif
170 
171  // Create new HttpRequest object if necessary
172  if (!currentRequest) {
173  currentRequest=new HttpRequest(settings);
174  }
175 
176  // Collect data for the request object
177  while (socket.bytesAvailable() && currentRequest->getStatus()!=HttpRequest::complete && currentRequest->getStatus()!=HttpRequest::abort) {
178  currentRequest->readFromSocket(socket);
179  if (currentRequest->getStatus()==HttpRequest::waitForBody) {
180  // Restart timer for read timeout, otherwise it would
181  // expire during large file uploads.
182  int readTimeout=settings->value("readTimeout",10000).toInt();
183  readTimer.start(readTimeout);
184  }
185  }
186 
187  // If the request is aborted, return error message and close the connection
188  if (currentRequest->getStatus()==HttpRequest::abort) {
189  socket.write("HTTP/1.1 413 entity too large\r\nConnection: close\r\n\r\n413 Entity too large\r\n");
190  socket.disconnectFromHost();
191  delete currentRequest;
192  currentRequest=0;
193  return;
194  }
195 
196  // If the request is complete, let the request mapper dispatch it
197  if (currentRequest->getStatus()==HttpRequest::complete) {
198  readTimer.stop();
199  qDebug("HttpConnectionHandler (%p): received request",this);
200  HttpResponse response(&socket);
201  try {
202  requestHandler->service(*currentRequest, response);
203  }
204  catch (...) {
205  qCritical("HttpConnectionHandler (%p): An uncatched exception occured in the request handler",this);
206  }
207 
208  // Finalize sending the response if not already done
209  if (!response.hasSentLastPart()) {
210  response.write(QByteArray(),true);
211  }
212  // Close the connection after delivering the response, if requested
213  if (QString::compare(currentRequest->getHeader("Connection"),"close",Qt::CaseInsensitive)==0) {
214  socket.disconnectFromHost();
215  }
216  else {
217  // Start timer for next request
218  int readTimeout=settings->value("readTimeout",10000).toInt();
219  readTimer.start(readTimeout);
220  }
221  // Prepare for next request
222  delete currentRequest;
223  currentRequest=0;
224  }
225  }
